# Python SDK

The official Authlete Python SDK provides a comprehensive client library for integrating OAuth 2.0 and OpenID Connect functionality into your Python applications.

## Installation

```bash
pip install authlete-python
```

## Quick Start

```python
from authlete import AuthleteApi
from authlete.dto import Service, ServiceCreateRequest

# Initialize the API client
api = AuthleteApi(
    base_url="https://api.authlete.com,
    service_api_key="your-service-api-key",
    service_api_secret="your-service-api-secret
)

# Create a service
service = Service()
service.service_name = "My OAuth Service"
service.supported_grant_types = ["AUTHORIZATION_CODE", "REFRESH_TOKEN"]

request = ServiceCreateRequest()
request.service = service

created_service = api.create_service(request)
```

## Examples

### Authorization Code Flow

```python
from authlete.dto import AuthorizationRequest, AuthorizationResponse

# Process authorization request
request = AuthorizationRequest()
request.parameters = 

response = api.process_authorization_request(request)

if response.action == "INTERACTION":
    # Show authorization page
    # After user consent, call issue_authorization_response
    from authlete.dto import AuthorizationIssueRequest
    
    issue_request = AuthorizationIssueRequest()
    issue_request.ticket = response.ticket
    issue_request.subject = "user123"
    
    issue_response = api.issue_authorization_response(issue_request)
```

### Async Example

```python
import asyncio
from authlete import AsyncAuthleteApi

async def main():
    api = AsyncAuthleteApi(
        base_url="https://api.authlete.com,
        service_api_key="your-service-api-key",
        service_api_secret="your-service-api-secret
    )
    
    # Create a service asynchronously
    service = Service()
    service.service_name = "My Async OAuth Service"
    
    request = ServiceCreateRequest()
    request.service = service
    
    created_service = await api.create_service(request)
    print(f"Created service: ")

# Run the async function
asyncio.run(main())
```

### Flask Integration

```python
from flask import Flask, request, jsonify
from authlete import AuthleteApi

app = Flask(__name__)
api = AuthleteApi(
    base_url="https://api.authlete.com,
    service_api_key="your-service-api-key",
    service_api_secret="your-service-api-secret
)

@app.route('/authorize')
  def authorize():
    auth_request = AuthorizationRequest()
    auth_request.parameters = request.args.to_dict()
    
    response = api.process_authorization_request(auth_request)
    
    if response.action == "INTERACTION":
        return jsonify()
    
    return jsonify(response.to_dict())

if __name__ == '__main__':
    app.run()
```
