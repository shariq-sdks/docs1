## Prerequisites

Before you begin, make sure you have completed the following. See [Getting Started with Authlete](/getting-started) for detailed instructions:

1. **Sign up for an Authlete account** - Create your account at [console.authlete.com](https://console.authlete.com/register)

2. **Create a service** - Set up your OAuth 2.0 authorization server service in the Authlete Console

3. **Register a client application** - Create a client application that will use your authorization server

Once you have your Service ID and Bearer Token (Service Access Token) from the Authlete Console, you're ready to implement the OAuth endpoints!

## OAuth 2.0 Flow

This quickstart implements the OAuth 2.0 authorization code flow with four endpoints:

1. **Authorization endpoint** - Receives authorization requests from clients and returns a ticket for user consent
2. **Authorization issue endpoint** - Processes user consent and issues an authorization code
3. **Token endpoint** - Exchanges the authorization code for access tokens
4. **Introspection endpoint** - Validates access tokens for protected API access

<Info>
See [How Authlete Works](/overview) to understand the OAuth 2.0 flow and what Authlete does vs what you do.
</Info>

## Implementation Steps

<Steps>
  <Step title="Download and initialize SDK">
    Install the Authlete TypeScript SDK. This package provides all the methods you need to implement OAuth 2.0 endpoints.

<CodeGroup>
    ```bash npm
    npm install @authlete/typescript-sdk
    ```
    </CodeGroup>
  </Step>

  <Step title="Download project structure">
    Download the starter project that includes Express server setup, TypeScript configuration, and all necessary files. It will create an app with the name `my-auth-server` in your current directory.

    <CodeGroup>
    ```bash npm
    npm create authlete-app@latest my-auth-server
    cd my-auth-server
    ```

    ```bash npx
    npx create-authlete-app@latest my-auth-server
    cd my-auth-server
    ```
    </CodeGroup>
  </Step>

  <Step title="Environment setup">
    Configure your environment variables with your Authlete credentials. This tutorial assumes you have already created a service and client as described in the [Getting Started with Authlete](/getting-started) guide.

    | Item | Value |
    |------|-------|
    | Authlete API - US Cluster | `https://us.authlete.com` |
    | Service ID | Auto-generated, e.g., `933860280` |
    | Service Access Token (Bearer Token) | Auto-generated, e.g., `DL7jo1z3-iUIXyI5MnX...` |
    | Client ID | Auto-generated, e.g., `12818600553323` |
    | Client Secret | Auto-generated |

    Open the `.env` file in your project root and replace the sample values:

    <CodeGroup>
    ```env .env
    AUTHLETE_BASE_URL=https://us.authlete.com
    AUTHLETE_SERVICE_ID=your_service_id
    AUTHLETE_BEARER_TOKEN=your_bearer_token
    PORT=3000
```
</CodeGroup>

    <Info>
    The `AUTHLETE_BEARER_TOKEN` is your Service Access Token from the Authlete Console. The Service Access Token and Bearer Token are the same credential.
    </Info>

    <Warning>
    Get your credentials from the [Authlete Console](https://console.authlete.com). Never commit your `.env` file to version control.
    </Warning>
  </Step>

  <Step title="Implement authorization endpoint">
    Open `src/endpoints/authorization.ts` and paste this code. This endpoint receives authorization requests from clients, validates them with Authlete, and returns a ticket that will be used in the next step for user consent.

<CodeGroup>
    ```typescript TypeScript
    import { Authlete } from '@authlete/typescript-sdk';
    import { Request, Response } from 'express';

    const authlete = new Authlete({
      bearer: process.env.AUTHLETE_BEARER_TOKEN || '',
    });

    export async function handleAuthorization(req: Request, res: Response) {
      const params = new URLSearchParams({
        response_type: req.body.response_type || 'code',
        client_id: req.body.client_id,
        redirect_uri: req.body.redirect_uri,
        scope: req.body.scope || 'openid',
        state: req.body.state || '',
      });

      const response = await authlete.authorizationEndpoint.authAuthorizationApi({
        serviceId: process.env.AUTHLETE_SERVICE_ID as string,
        requestBody: {
          parameters: params.toString(),
        },
      });

      if (response.action === 'INTERACTION') {
        return res.json({ ticket: response.ticket });
      }

      return res.status(400).json({ error: response.resultMessage });
    }
    ```

    ```bash curl
    curl -X POST "https://us.authlete.com/api/{SERVICE_ID}/auth/authorization" \
      -H "Authorization: Bearer $AUTHLETE_BEARER_TOKEN" \
  -H "Content-Type: application/json" \
      -d '{
        "parameters": "response_type=code&client_id=YOUR_CLIENT_ID&redirect_uri=https://client.example.org/callback&scope=openid"
      }'
    ```
    </CodeGroup>
  </Step>

  <Step title="Implement authorization issue endpoint">
    Open `src/endpoints/authorization-issue.ts` and paste this code. This endpoint processes the user's consent decision using the ticket from the previous step and issues an authorization code that the client will exchange for tokens.

    <CodeGroup>
    ```typescript TypeScript
    import { Authlete } from '@authlete/typescript-sdk';
    import { Request, Response } from 'express';

    const authlete = new Authlete({
      bearer: process.env.AUTHLETE_BEARER_TOKEN || '',
    });

    export async function handleAuthorizationIssue(req: Request, res: Response) {
      const response = await authlete.authorizationEndpoint.authAuthorizationIssueApi({
        ticket: req.body.ticket,
        subject: 'user123', // Replace with actual authenticated user ID
        claims: {},
      });

      if (response.action === 'OK') {
        return res.json({
          authorizationCode: response.authorizationCode,
          redirectUri: response.responseContent,
        });
      }

      return res.status(400).json({ error: response.resultMessage });
    }
    ```

    ```bash curl
    curl -X POST "https://us.authlete.com/api/{SERVICE_ID}/auth/authorization/issue" \
      -H "Authorization: Bearer $AUTHLETE_BEARER_TOKEN" \
      -H "Content-Type: application/json" \
      -d '{
        "ticket": "TICKET_FROM_AUTHORIZATION_STEP",
        "subject": "user123"
      }'
    ```
    </CodeGroup>
  </Step>

  <Step title="Implement token endpoint">
    Open `src/endpoints/token.ts` and paste this code. This endpoint exchanges the authorization code (received from the client) for access tokens and refresh tokens that can be used to access protected resources.

    <CodeGroup>
    ```typescript TypeScript
    import { Authlete } from '@authlete/typescript-sdk';
    import { Request, Response } from 'express';

    const authlete = new Authlete({
      bearer: process.env.AUTHLETE_BEARER_TOKEN || '',
    });

    export async function handleToken(req: Request, res: Response) {
      const params = new URLSearchParams({
        grant_type: req.body.grant_type || 'authorization_code',
        code: req.body.code,
        client_id: req.body.client_id,
        client_secret: req.body.client_secret || '',
        redirect_uri: req.body.redirect_uri,
      });

      const response = await authlete.tokenEndpoint.postTokenApi({
        parameters: params.toString(),
      });

      if (response.action === 'OK') {
        return res.json({
          access_token: response.accessToken,
          token_type: 'Bearer',
          expires_in: response.accessTokenExpiresAt,
          refresh_token: response.refreshToken,
          id_token: response.idToken,
          scope: response.scope,
        });
      }

      return res.status(400).json({ 
        error: response.error,
        error_description: response.errorDescription 
      });
    }
    ```

    ```bash curl
    curl -X POST "https://us.authlete.com/api/{SERVICE_ID}/auth/token" \
      -H "Authorization: Bearer $AUTHLETE_BEARER_TOKEN" \
      -H "Content-Type: application/json" \
      -d '{
        "clientId": "YOUR_CLIENT_ID",
        "clientSecret": "YOUR_CLIENT_SECRET",
        "parameters": "grant_type=authorization_code&code=AUTHORIZATION_CODE&redirect_uri=https://client.example.org/callback"
      }'
    ```
    </CodeGroup>
  </Step>

  <Step title="Implement introspection endpoint">
    Open `src/endpoints/introspection.ts` and paste this code. This endpoint validates access tokens when clients make API requests. Use this in your resource server to verify tokens before allowing access to protected resources.

    <CodeGroup>
    ```typescript TypeScript
    import { Authlete } from '@authlete/typescript-sdk';
    import { Request, Response } from 'express';

    const authlete = new Authlete({
      bearer: process.env.AUTHLETE_BEARER_TOKEN || '',
    });

    export async function handleIntrospection(req: Request, res: Response) {
      const response = await authlete.introspectionEndpoint.postIntrospectionApi({
        token: req.body.token,
      });

      if (response.active) {
        return res.json({
          active: true,
          scope: response.scope,
          client_id: response.clientId,
          username: response.username,
          subject: response.subject,
          exp: response.expiresAt,
        });
      }

      return res.json({ active: false });
    }
    ```

    ```bash curl
    curl -X POST "https://us.authlete.com/api/{SERVICE_ID}/auth/introspection" \
      -H "Authorization: Bearer $AUTHLETE_BEARER_TOKEN" \
      -H "Content-Type: application/json" \
      -d '{
        "token": "ACCESS_TOKEN_FROM_TOKEN_STEP"
      }'
    ```
    </CodeGroup>
  </Step>

  <Step title="Test your authorization server">
    Start your server by running the following command. This will start the Express server on port 3000.

    <CodeGroup>
    ```bash
    npm run dev
```
</CodeGroup>

    Visit your server at [http://localhost:3000](http://localhost:3000). Your authorization server is now running with four endpoints:

    - `POST /auth/authorization` - Push authorization request to Authlete server
    - `POST /auth/authorization/issue` - Process user consent and issue authorization code
    - `POST /auth/token` - Exchange authorization code for access token
    - `POST /auth/introspection` - Validate access tokens and get token metadata
  </Step>

  </Step>
</Steps>
